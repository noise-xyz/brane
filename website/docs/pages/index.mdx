```java [Connect]
// HD wallet with derived accounts (BIP-44)
var wallet = MnemonicWallet.fromPhrase(mnemonic);
var account0 = wallet.derive(0);  // m/44'/60'/0'/0/0
var account1 = wallet.derive(1);  // m/44'/60'/0'/0/1
var account2 = wallet.derive(2);  // m/44'/60'/0'/0/2

// Connect with signing capabilities
var signer = Brane.connect(rpcUrl, account0);

// Check connection and chain
var chainId = signer.getChainId();
var balance = signer.getBalance(account0.address());
log.info("Connected to chain {} with balance {}", chainId, balance.toEther());

// WebSocket with real-time block subscriptions
var ws = Brane.connectWs("wss://eth.llamarpc.com");
ws.subscribeNewHeads(block -> {
    log.info("New block: {} with {} txs", block.number(), block.transactions().size());
    processBlock(block);
});

// Subscribe to contract events
ws.subscribeLogs(usdcAddress, transferTopic, log -> {
    var transfer = decodeTransfer(log);
    log.info("Transfer: {} -> {} amount {}", transfer.from(), transfer.to(), transfer.amount());
});
```

```java [Read]
// Define contract interface - no codegen required
interface ERC20 {
    BigInteger balanceOf(Address owner);
    BigInteger allowance(Address owner, Address spender);
    BigInteger totalSupply();
    int decimals();
    String symbol();
}

// Bind interface to deployed contract
var usdc = BraneContract.bind(ERC20.class, abi, usdcAddress, client);

// Single call
var balance = usdc.balanceOf(wallet);
var decimals = usdc.decimals();
log.info("Balance: {} {}", formatUnits(balance, decimals), usdc.symbol());

// Multicall batching - single RPC call for multiple reads
var state = Multicall.execute(client, () -> {
    return new TokenState(
        usdc.balanceOf(wallet),
        usdc.allowance(wallet, spender),
        usdc.totalSupply(),
        usdc.decimals()
    );
});
log.info("Fetched {} values in single RPC call", 4);
```

```java [Write]
// EIP-4844 blob transaction for L2 data availability
var kzg = CKzg.loadFromClasspath();

// Build blob transaction with rollup batch data
var request = Eip4844Builder.create()
    .to(blobRecipient)
    .blobData(rollupBatchData)  // auto-encoded into blobs with KZG proofs
    .maxFeePerBlobGas(Wei.fromGwei("100"))
    .build(kzg);

// Verify blob count before sending
log.info("Sending {} blobs ({} bytes)",
    request.sidecar().blobs().size(),
    rollupBatchData.length);

// Send and wait for confirmation
var receipt = signer.sendBlobTransactionAndWait(request);
log.info("Blob tx confirmed in block {}", receipt.blockNumber());
log.info("Gas used: {} | Blob gas used: {}", receipt.gasUsed(), receipt.blobGasUsed());

// Decode original data from blobs for verification
var decoded = BlobDecoder.decode(request.sidecar().blobs());
assert Arrays.equals(decoded, rollupBatchData);
```

```java [Test]
// Connect to local Anvil instance
var tester = Brane.connectTest();

// Snapshot state for test isolation
var snapshot = tester.snapshot();

// Fund test accounts
var testAccount = new Address("0x1234...");
tester.setBalance(testAccount, Wei.fromEther(new java.math.BigDecimal("1000")));

// Impersonate any address without private key
var whaleAddress = new Address("0xWhale...");
try (var session = tester.impersonate(whaleAddress)) {
    // Execute transactions as whale
    var receipt = session.sendTransactionAndWait(
        Eip1559Builder.create()
            .to(recipient)
            .value(Wei.fromEther(new java.math.BigDecimal("100")))
            .build()
    );
    log.info("Whale transfer confirmed: {}", receipt.transactionHash());
}

// Time travel for vesting/lock testing
tester.increaseTime(Duration.ofDays(30));
tester.mine();

// Verify time-locked funds are now accessible
var unlockedBalance = vestingContract.releasable(testAccount);
log.info("Unlocked after 30 days: {}", unlockedBalance);

// Revert to clean state
tester.revert(snapshot);
```
